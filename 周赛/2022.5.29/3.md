class Solution {
public:
    int totalSteps(vector<int>& nums) 
    {
        bool sorted = false;
        int count = 0;
        while(!sorted)
        {
            int k = 1;
            while(k < nums.size())
            {
                if(nums[k - 1] > nums[k])
                    break;
                k ++;
            }
            
            if(k == nums.size())
            {
                sorted = true;
                break;
            }
            
            count ++;
            
            vector<int> p;
            for(int i = 1 ; i < nums.size() ; i ++)
                if(nums[i - 1] > nums[i])
                    p.push_back(i);
            
            int n = p.size();
            while(n)
            {
                nums.erase(nums.begin() + p[n - 1]);
                n --;
            }
            
            for(int i = 0 ; i < nums.size() ; i ++)
                cout << nums[i] << ' ';
        }
          
        return count;
    }
};
